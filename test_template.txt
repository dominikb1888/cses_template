#[test]
fn test_{test_name}() {
    let input = r#"{input_content}"#;
    let expected_output = r#"{expected_output}"#;

    // Create a buffer to capture the output
    let mut buffer = Vec::new();
    
    // Call the function passing the buffer as the writer
    process(input, &mut buffer);

    let output = String::from_utf8(buffer).expect("Output was not valid UTF-8");

    assert_eq!(
        output.trim(),
        expected_output.trim(),
        "Test failed for \"{}\".\nExpected:\n{}\nGot:\n{}",
        "{input_file}",
        expected_output.trim(),
        output.trim()
    );
}
